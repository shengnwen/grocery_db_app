create table user_(
     user_ID       int,
     zipcode       int,
     password      varchar(256),
     email         varchar(254),
     primary key (user_ID)
);

create table store(
     store_ID      int,
     name          varchar(80),
     address       varchar(80),
     gps_x         numeric(7, 4),
     gps_y         numeric(7, 4),
     website       varchar(2048),
     primary key (store_ID)
);

create table food_type(
     food_type_name varchar(100),
     primary key (food_type_name)
);

create table product(
     name           varchar(100),
     brand          varchar(100),
     food_type_name varchar(100),
     primary key (name),
     foreign key (food_type_name) references food_type
);

create table stocks(
     store_ID      int,
     name          varchar(100),
     price         numeric(10, 2),
     url           varchar(2048),
     primary key (store_ID, name),
     foreign key (store_ID) references store,
     foreign key (name) references product
);
